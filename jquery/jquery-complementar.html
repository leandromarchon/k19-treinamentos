Ready

            $(document).ready(function(){
                var saida = $("#saida");
                saida.append("Evento ready disparado<br>");
            });

Resize

            /* ON */
            $(window).resize(function(){
                var largura = $(window).width();
                var altura = $(window).height();
                
                saida.append("Resize: (" + largura + ", " + altura + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $(window).off("resize");

Scroll

            /* ON */
            $(window).scroll(function(){
                var x = $(window).scrollLeft();
                var y = $(window).scrollTop();

                saida.append("Scroll: (" + x + ", " + y + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $(window).off("scroll");

Focus, FocusIn, Blur e FocusOut

            /* ON */
            $("*").on(focus focusin blur focusout, function(event){
                var tag = this.tagName;
                var id = this.id;
                var type = event.type;

                saida.append(type + "(" + tag + ", #" + id + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $(window).off("focus focusin blur focusout");

Select

            /* ON */
            $("#input2").select(function(){
                saida.append("Select: (" + window.getSelection() + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $("#input2").off("select");

Change

            /* ON */
            $("#input3").change(function(){
                saida.append("Change: (" + $(this).val() + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $("#input3").off("change");

Keydown, Keypress e KeyUp

            /* ON */
            $("#input4").on("keydown keypress keyup", function(event){
                var type = event.type;

                saida.append(type + "(" + event.which + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $("#input4").off("keydown keypress keyup");

Click, DblClick, MouseDown, MouseUp, MouseEnter, MouseLeave, Hover, MouseOver e MouseOut

            /* ON */
            $("#div1").on("click dblclick mousedown mouseup mouseenter mouseleaver hover mouseover mouseout", function(event){
                var type = event.type;

                saida.append(type + "(" + event.which + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $("#div1").off("click dblclick mousedown mouseup mouseenter mouseleaver hover mouseover mouseout");

Mousemove

            /* ON */
            $("#div2").mousemove(function(event){
                var x = event.pageX;
                var y = event.pageY;
                var type = event.type;

                saida.append(type + "(" + x + ", " + y + ")");
                saida.append("<br>");
                saida.scrollTop(saida.prop("scrollHeight"));
            });

            /* OFF */
            $("#div2").off("mousemove");
